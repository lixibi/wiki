```python
def w_day(y, m, d):  # 已知日期求星期
    c = y // 100
    y = int(str(y)[-2:])
    w = (y + (y // 4) + (c // 4) - 2 * c + (26 * (m + 1) // 10) + d - 1) % 7
    ##蔡勒公式只适合于1582年10月15日之后
    return w
```
